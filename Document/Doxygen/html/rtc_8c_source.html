48 int RTC_GetWeekFromYMD(int year, int month, int days)
49 {
50  static int mdays[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30 };
51  int i, y = year - 1;

67 static void RTC_DateTimeToSecond(const RTC_DateTime_Type * datetime, uint32_t * seconds)
68 {
69  /* Compute number of days from 1970 till given year*/
70  *seconds = (datetime->year - 1970U) * DAYS_IN_A_YEAR;
71  /* Add leap year days */
72  *seconds += ((datetime->year / 4) - (1970U / 4));
73  /* Add number of days till given month*/
74  *seconds += MONTH_DAYS[datetime->month];
75  /* Add days in given month*/
76  *seconds += datetime->day;
77  /* For leap year if month less than or equal to Febraury, decrement day counter*/
78  if ((!(datetime->year & 3U)) && (datetime->month <= 2U))
79  {
80  (*seconds)--;
81  }
82 
83  *seconds = ((*seconds) * SECONDS_IN_A_DAY) + (datetime->hour * SECONDS_IN_A_HOUR) +
84  (datetime->minute * SECONDS_IN_A_MIN) + datetime->second;
85  (*seconds)++;
86 }

94 static void RTC_SecondToDateTime(const uint32_t * seconds, RTC_DateTime_Type * datetime)
95 {
96  uint32_t x;
97  uint32_t Seconds, Days, Days_in_year;
98  const uint8_t *Days_in_month;
99  /* Start from 1970-01-01*/
100  Seconds = *seconds;
101  /* days*/
102  Days = Seconds / SECONDS_IN_A_DAY;
103  /* seconds left*/
104  Seconds = Seconds % SECONDS_IN_A_DAY;
105  /* hours*/
106  datetime->hour = Seconds / SECONDS_IN_A_HOUR;
107  /* seconds left*/
108  Seconds = Seconds % SECONDS_IN_A_HOUR;
109  /* minutes*/
110  datetime->minute = Seconds / SECONDS_IN_A_MIN;
111  /* seconds*/
112  datetime->second = Seconds % SECONDS_IN_A_MIN;
113  /* year*/
114  datetime->year = 1970;
115  Days_in_year = DAYS_IN_A_YEAR;
116 
117  while (Days > Days_in_year)
118  {
119  Days -= Days_in_year;
120  datetime->year++;
121  if (datetime->year & 3U)
122  {
123  Days_in_year = DAYS_IN_A_YEAR;
124  }
125  else
126  {
127  Days_in_year = DAYS_IN_A_LEAP_YEAR;
128  }
129  }
130 
131  if (datetime->year & 3U)
132  {
133  Days_in_month = ULY;
134  }
135  else
136  {
137  Days_in_month = LY;
138  }
139 
140  for (x=1U; x <= 12U; x++)
141  {
142  if (Days <= (*(Days_in_month + x)))
143  {
144  datetime->month = x;
145  break;
146  }
147  else
148  {
149  Days -= (*(Days_in_month + x));
150  }
151  }
152  datetime->day = Days;
153 }
