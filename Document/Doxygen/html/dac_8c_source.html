49 void DAC_Init(DAC_InitTypeDef* DAC_InitStruct)
50 {
51  SIM->SCGC2 |= SIM_SCGC2_DAC0_MASK;
52  /* trigger select */
53  switch(DAC_InitStruct->triggerMode)
54  {
55  case kDAC_TriggerSoftware:
56  DAC_InstanceTable[DAC_InitStruct->instance]->C0 |= DAC_C0_DACTRGSEL_MASK;
57  break;
58  case kDAC_TriggerHardware:
59  DAC_InstanceTable[DAC_InitStruct->instance]->C0 &= ~DAC_C0_DACTRGSEL_MASK;
60  break;
61  default:
62  break;
63  }
64  /* buffer mode */
65  switch(DAC_InitStruct->bufferMode)
66  {
67  case kDAC_Buffer_Disable:
68  DAC_InstanceTable[DAC_InitStruct->instance]->C1 &= ~DAC_C1_DACBFEN_MASK;
69  break;
70  case kDAC_Buffer_Normal:
71  DAC_InstanceTable[DAC_InitStruct->instance]->C1 |= DAC_C1_DACBFEN_MASK;
72  DAC_InstanceTable[DAC_InitStruct->instance]->C1 &= ~DAC_C1_DACBFMD_MASK;
73  DAC_InstanceTable[DAC_InitStruct->instance]->C1 |= DAC_C1_DACBFMD(0);
74  break;
75  case kDAC_Buffer_Swing:
76  DAC_InstanceTable[DAC_InitStruct->instance]->C1 |= DAC_C1_DACBFEN_MASK;
77  DAC_InstanceTable[DAC_InitStruct->instance]->C1 &= ~DAC_C1_DACBFMD_MASK;
78  DAC_InstanceTable[DAC_InitStruct->instance]->C1 |= DAC_C1_DACBFMD(1);
79  break;
80  case kDAC_Buffer_OneTimeScan:
81  DAC_InstanceTable[DAC_InitStruct->instance]->C1 |= DAC_C1_DACBFEN_MASK;
82  DAC_InstanceTable[DAC_InitStruct->instance]->C1 &= ~DAC_C1_DACBFMD_MASK;
83  DAC_InstanceTable[DAC_InitStruct->instance]->C1 |= DAC_C1_DACBFMD(2);
84  break;
85  default:
86  break;
87  }
88  /* reference */
89  switch(DAC_InitStruct->referenceMode)
90  {
91  case kDAC_Reference_1:
92  DAC_InstanceTable[DAC_InitStruct->instance]->C0 &= ~DAC_C0_DACRFS_MASK;
93  break;
94  case kDAC_Reference_2:
95  DAC_InstanceTable[DAC_InitStruct->instance]->C0 |= DAC_C0_DACRFS_MASK;
96  break;
97  default:
98  break;
99  }
100  /* clear flags */
101  DAC_InstanceTable[DAC_InitStruct->instance]->SR &= ~DAC_SR_DACBFRPBF_MASK;
102  DAC_InstanceTable[DAC_InitStruct->instance]->SR &= ~DAC_SR_DACBFRPTF_MASK;
103  DAC_InstanceTable[DAC_InitStruct->instance]->SR &= ~DAC_SR_DACBFWMF_MASK;
104  /* enable module */
105  DAC_InstanceTable[DAC_InitStruct->instance]->C0 |= DAC_C0_DACEN_MASK;
106 }

171 {
172  switch(config)

174  case kDAC_IT_Disable:
175  NVIC_DisableIRQ(DAC_IRQnTable[instance]);
176  break;
177  case kDAC_DMA_Disable:
178  DAC_InstanceTable[instance]->C1 &= ~DAC_C1_DMAEN_MASK;
179  break;
180  case kDAC_IT_BufferPointer_WaterMark:
181  DAC_InstanceTable[instance]->C0 |= DAC_C0_DACBWIEN_MASK;
182  NVIC_EnableIRQ(DAC_IRQnTable[instance]);
183  break;
184  case kDAC_IT_BufferPointer_TopFlag:
185  DAC_InstanceTable[instance]->C0 |= DAC_C0_DACBTIEN_MASK;
186  NVIC_EnableIRQ(DAC_IRQnTable[instance]);
187  break;

189  DAC_InstanceTable[instance]->C0 |= DAC_C0_DACBBIEN_MASK;
190  NVIC_EnableIRQ(DAC_IRQnTable[instance]);
191  break;
192  case kDAC_DMA_BufferPointer_WaterMark:
193  DAC_InstanceTable[instance]->C0 |= DAC_C0_DACBWIEN_MASK;
194  DAC_InstanceTable[instance]->C1 |= DAC_C1_DMAEN_MASK;
195  break;

197  DAC_InstanceTable[instance]->C0 |= DAC_C0_DACBTIEN_MASK;

302 static void DAC_IRQHandler(uint32_t instance)
303 {
304  /* clear IT pending bit */
305  uint32_t C0 = DAC_InstanceTable[instance]->C0;
306  if(C0 & DAC_C0_DACBWIEN_MASK)
307  {
308  DAC_InstanceTable[instance]->SR &= ~DAC_SR_DACBFWMF_MASK;
309  }
310  if(C0 & DAC_C0_DACBBIEN_MASK)
311  {
312  DAC_InstanceTable[instance]->SR &= ~DAC_SR_DACBFRPBF_MASK;
313  }
314  if(C0 & DAC_C0_DACBTIEN_MASK)
315  {
316  DAC_InstanceTable[instance]->SR &= ~DAC_SR_DACBFRPTF_MASK;
317  }
318  /* call back */
319  if(DAC_CallBackTable[instance] != NULL)
320  {
321  DAC_CallBackTable[instance]();
322  }
323 }
