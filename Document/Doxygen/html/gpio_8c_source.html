77 void PORT_PinPullConfig(uint32_t instance, uint8_t pin, PORT_Pull_Type pull)
78 {
79  SIM->SCGC5 |= SIM_GPIOClockGateTable[instance];
80  switch(pull)
81  {
82  case kPullDisabled:
83  PORT_InstanceTable[instance]->PCR[pin] &= ~PORT_PCR_PE_MASK;
84  break;
85  case kPullUp:
86  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_PE_MASK;
87  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_PS_MASK;
88  break;
89  case kPullDown:
90  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_PE_MASK;
91  PORT_InstanceTable[instance]->PCR[pin] &= ~PORT_PCR_PS_MASK;
92  break;
93  default:
94  break;

185 void GPIO_Init(GPIO_InitTypeDef * GPIO_InitStruct)
186 {
187  /* config state */
188  switch(GPIO_InitStruct->mode)
189  {
190  case kGPIO_Mode_IFT:
191  PORT_PinPullConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kPullDisabled);
192  PORT_PinOpenDrainConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, DISABLE);
193  GPIO_PinConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kInput);
194  break;
195  case kGPIO_Mode_IPD:
196  PORT_PinPullConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kPullDown);
197  PORT_PinOpenDrainConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, DISABLE);
198  GPIO_PinConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kInput);
199  break;
200  case kGPIO_Mode_IPU:
201  PORT_PinPullConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kPullUp);
202  PORT_PinOpenDrainConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, DISABLE);
203  GPIO_PinConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kInput);
204  break;
205  case kGPIO_Mode_OOD:
206  PORT_PinPullConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kPullUp);
207  PORT_PinOpenDrainConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, ENABLE);
208  GPIO_PinConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kOutput);
209  break;
210  case kGPIO_Mode_OPP:
211  PORT_PinPullConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kPullDisabled);
212  PORT_PinOpenDrainConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, DISABLE);
213  GPIO_PinConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kOutput);
214  break;
215  default:
216  break;
217  }
218  /* config pinMux */
219  PORT_PinMuxConfig(GPIO_InitStruct->instance, GPIO_InitStruct->pinx, kPinAlt1);
220 }

437 void GPIO_ITDMAConfig(uint32_t instance, uint8_t pin, GPIO_ITDMAConfig_Type config, bool status)
438 {
439  /* init moudle */
440  SIM->SCGC5 |= SIM_GPIOClockGateTable[instance];
441  PORT_InstanceTable[instance]->PCR[pin] &= ~PORT_PCR_IRQC_MASK;
442 
443  if(!status)
444  {
445  NVIC_DisableIRQ(GPIO_IRQnTable[instance]);
446  return;
447  }
448 
449  switch(config)
450  {
451  case kGPIO_DMA_RisingEdge:
452  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_IRQC(1);
453  break;
454  case kGPIO_DMA_FallingEdge:
455  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_IRQC(2);
456  break;
457  case kGPIO_DMA_RisingFallingEdge:
458  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_IRQC(3);
459  break;
460  case kGPIO_IT_Low:
461  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_IRQC(8);
462  NVIC_EnableIRQ(GPIO_IRQnTable[instance]);
463  break;
464  case kGPIO_IT_RisingEdge:
465  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_IRQC(9);
466  NVIC_EnableIRQ(GPIO_IRQnTable[instance]);
467  break;
468  case kGPIO_IT_FallingEdge:
469  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_IRQC(10);
470  NVIC_EnableIRQ(GPIO_IRQnTable[instance]);
471  break;
472  case kGPIO_IT_RisingFallingEdge:
473  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_IRQC(11);
474  NVIC_EnableIRQ(GPIO_IRQnTable[instance]);
475  break;
476  case kGPIO_IT_High:
477  PORT_InstanceTable[instance]->PCR[pin] |= PORT_PCR_IRQC(12);
478  NVIC_EnableIRQ(GPIO_IRQnTable[instance]);
479  break;
480  default:
481  break;
482  }
