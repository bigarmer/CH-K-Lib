164 void NFC_SendResetCmd(void)
165 {
166  /* Clear all status and error bits in the NFC_ISR register */
167  NFC->ISR |= ( NFC_ISR_WERRCLR_MASK
168  | NFC_ISR_DONECLR_MASK
169  | NFC_ISR_IDLECLR_MASK );
170 
171  /* Write the NFC_CMD2 register with the command byte and code for a reset */
172  NFC->CMD2 = NFC_CMD2_BYTE1(RESET_CMD_BYTE) | NFC_CMD2_CODE(NFC_RESET_CMD_CODE);
173 
174  /* Set Start Bit to send reset to the NAND flash */
175  NFC->CMD2 |= NFC_CMD2_BUSY_START_MASK;
176 
177  /* Wait for start/busy bit to clear indicating command is done */
178  while (NFC->CMD2 & NFC_CMD2_BUSY_START_MASK);
179 }

228

230  NFC->CFG &= ~NFC_CFG_ECCMODE(0x7);
231 
232  /* Make sure the column address is cleared - not needed for block erase */
233  NFC->CAR = 0x0;
234

236  switch(cs)

242  NFC->RAR = (NFC_RAR_CS1_MASK | NFC_RAR_RB1_MASK);
243  break;
244  default:
245  break;
246  }
247

249  NFC->RAR |= row_addr;

253  | NFC_CMD2_CODE(NFC_BLOCK_ERASE_CMD_CODE) );
