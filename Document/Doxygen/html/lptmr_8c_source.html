50 void LPTMR_TC_Init(LPTMR_TC_InitTypeDef* LPTMR_TC_InitStruct)
51 {
52  /* open clock gate */
53  *(uint32_t*)SIM_LPTMRClockGateTable[0].addr |= SIM_LPTMRClockGateTable[0].mask;
54 
55  LPTMR0->CSR = 0x00;
56  LPTMR0->PSR = 0x00;
57  LPTMR0->CMR = 0x00;
58 
59  /* disable module first */
60  LPTMR0->CSR &= ~LPTMR_CSR_TEN_MASK;
61 
62  /* free counter will reset whenever compare register is writtened. */
63  LPTMR0->CSR &= ~LPTMR_CSR_TFC_MASK;
64 
65  /* timer counter mode */
66  LPTMR0->CSR &= ~LPTMR_CSR_TMS_MASK;
67 
68  /* bypass the prescaler, which mean we use 1KHZ LPO directly */
69  LPTMR0->PSR = LPTMR_PSR_PCS(1)| LPTMR_PSR_PBYP_MASK;
70

72  LPTMR0->CMR = LPTMR_CMR_COMPARE(LPTMR_TC_InitStruct->timeInMs);
73 
74  /* enable moudle */
75  LPTMR0->CSR |= LPTMR_CSR_TEN_MASK;

92 void LPTMR_PC_Init(LPTMR_PC_InitTypeDef* LPTMR_PC_InitStruct)

96 
97  LPTMR0->CSR = 0x00;
98  LPTMR0->PSR = 0x00;
99  LPTMR0->CMR = 0x00;
100 
101  /* disable module first */
102  LPTMR0->CSR &= ~LPTMR_CSR_TEN_MASK;
103 
104  /* free counter will reset whenever compare register is writtened. */
105  LPTMR0->CSR &= ~LPTMR_CSR_TFC_MASK;
106 
107  /* timer counter mode */
108  LPTMR0->CSR |= LPTMR_CSR_TMS_MASK;
109 
110  /* bypass the glitch filter, which mean we use 1KHZ LPO directly */
111  LPTMR0->PSR = LPTMR_PSR_PCS(1)| LPTMR_PSR_PBYP_MASK;
112 
113  /* set CMR(compare register) */
114  LPTMR0->CMR = LPTMR_CMR_COMPARE(LPTMR_PC_InitStruct->counterOverflowValue);
115 
116  /* input source */
117  switch(LPTMR_PC_InitStruct->inputSource)
118  {
119  case kLPTMR_PC_InputSource_CMP0:
120  LPTMR0->CSR |= LPTMR_CSR_TPS(0);
121  break;
122  case kLPTMR_PC_InputSource_ALT1:
123  LPTMR0->CSR |= LPTMR_CSR_TPS(1);
124  break;
125  case kLPTMR_PC_InputSource_ALT2:
126  LPTMR0->CSR |= LPTMR_CSR_TPS(2);
127  break;
128  default:
129  break;
130  }
131 
132  /* pin polarity */
133  switch(LPTMR_PC_InitStruct->pinPolarity)
134  {
135  case kLPTMR_PC_PinPolarity_RigsingEdge:
136  LPTMR0->CSR &= ~LPTMR_CSR_TPP_MASK;
137  break;
138  case kLPTMR_PC_PinPolarity_FallingEdge:
139  LPTMR0->CSR |= LPTMR_CSR_TPP_MASK;
140  break;
141  default:
142  break;
143  }
144 
145  /* enable moudle */
146  LPTMR0->CSR |= LPTMR_CSR_TEN_MASK;
147 }
