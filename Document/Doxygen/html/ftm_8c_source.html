130 uint32_t FTM_QD_QuickInit(uint32_t MAP, FTM_QD_PolarityMode_Type polarity, FTM_QD_Mode_Type mode)

138  /* set FTM to QD mode */

141  /* QD mode config */
142  switch(polarity)

154  }
155  switch(mode)
156  {
157  case kQD_PHABEncoding:
158  FTM_InstanceTable[pq->ip]->QDCTRL &= ~FTM_QDCTRL_QUADMODE_MASK;
159  break;
160  case kQD_CountDirectionEncoding:
161  FTM_InstanceTable[pq->ip]->QDCTRL |= FTM_QDCTRL_QUADMODE_MASK;

163  default:
164  break;
165  }
166  /* init pinmux and pull up */
167  for(i = 0; i < pq->pin_cnt; i++)

169  PORT_PinMuxConfig(pq->io, pq->pin_start + i, (PORT_PinMux_Type) pq->mux);

249 static void FTM_DualChlConfig(uint32_t instance, uint8_t chl, FTM_DualChlConfig_Type mode, FunctionalState newState)
250 {
251  uint32_t mask = 0;
252  switch(mode)
253  {
254  case kFTM_Combine:
255  mask = FTM_COMBINE_COMBINE0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
256  break;
257  case kFTM_Complementary:
258  mask = FTM_COMBINE_COMP0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
259  break;
260  case kFTM_DualEdgeCapture:
261  mask = FTM_COMBINE_DECAPEN0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
262  break;
263  case kFTM_DeadTime:
264  mask = FTM_COMBINE_DTEN0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
265  break;
266  case kFTM_Sync:
267  mask = FTM_COMBINE_SYNCEN0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
268  break;
269  case kFTM_FaultControl:
270  mask = FTM_COMBINE_FAULTEN0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
271  break;
272  default:
273  break;
274  }
275  (newState == ENABLE)?(FTM_InstanceTable[instance]->COMBINE |= mask):(FTM_InstanceTable[instance]->COMBINE &= ~mask);
276 }

281 static void FTM_SetMode(uint32_t instance, uint8_t chl, FTM_Mode_Type mode)
282 {
283  switch(mode)
284  {
285  case kFTM_Mode_EdgeAligned:

288  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;
289  FTM_InstanceTable[instance]->CONTROLS[chl].CnSC |= (FTM_CnSC_MSB_MASK|FTM_CnSC_MSA_MASK);
290  FTM_PWM_InvertPolarity(instance, chl, kFTM_PWM_HighTrue);
291  FTM_DualChlConfig(instance, chl, kFTM_Combine, DISABLE);
292  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
293  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
294  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
295  FTM_DualChlConfig(instance, chl, kFTM_Sync, DISABLE);
296  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
297  break;
298  case kFTM_Mode_CenterAligned:
299  FTM_InstanceTable[instance]->MODE &= ~FTM_MODE_FTMEN_MASK;
300  FTM_InstanceTable[instance]->QDCTRL &= ~FTM_QDCTRL_QUADEN_MASK;
301  FTM_InstanceTable[instance]->SC |= FTM_SC_CPWMS_MASK;
302  FTM_DualChlConfig(instance, chl, kFTM_Combine, DISABLE);
303  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
304  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
305  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
306  FTM_DualChlConfig(instance, chl, kFTM_Sync, DISABLE);
307  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
308  break;
309  case kFTM_Mode_InputCapture:
310  /* all configuration on input capture */
311  FTM_InstanceTable[instance]->QDCTRL &= ~FTM_QDCTRL_QUADEN_MASK;
312  FTM_InstanceTable[instance]->MODE &= ~FTM_MODE_FTMEN_MASK;
313  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;
314  FTM_InstanceTable[instance]->CONTROLS[chl].CnSC &= ~(FTM_CnSC_MSB_MASK|FTM_CnSC_MSA_MASK);
315  FTM_DualChlConfig(instance, chl, kFTM_Combine, DISABLE);
316  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
317  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
318  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
319  FTM_DualChlConfig(instance, chl, kFTM_Sync, DISABLE);
320  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
321  break;
322  case kFTM_Mode_Complementary:
323  FTM_InstanceTable[instance]->QDCTRL &= ~FTM_QDCTRL_QUADEN_MASK;
324  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;
325  FTM_InstanceTable[instance]->MODE |= FTM_MODE_WPDIS_MASK;
326  FTM_InstanceTable[instance]->MODE |= FTM_MODE_FTMEN_MASK;
327  FTM_DualChlConfig(instance, chl, kFTM_Combine, ENABLE);
328  FTM_DualChlConfig(instance, chl, kFTM_Complementary, ENABLE);
329  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
330  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
331  FTM_DualChlConfig(instance, chl, kFTM_Sync, ENABLE);
332  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
333  FTM_InstanceTable[instance]->CONTROLS[chl].CnSC |= (FTM_CnSC_MSB_MASK|FTM_CnSC_MSA_MASK);
334  FTM_PWM_InvertPolarity(instance, chl, kFTM_PWM_HighTrue);
335  FTM_InstanceTable[instance]->SYNC = FTM_SYNC_CNTMIN_MASK|FTM_SYNC_CNTMAX_MASK;
336  FTM_InstanceTable[instance]->SYNC |= FTM_SYNC_SWSYNC_MASK;
337  break;
338  case kFTM_Mode_Combine:
339  FTM_InstanceTable[instance]->QDCTRL &= ~FTM_QDCTRL_QUADEN_MASK;
340  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;
341  FTM_InstanceTable[instance]->MODE |= FTM_MODE_WPDIS_MASK;
342  FTM_InstanceTable[instance]->MODE |= FTM_MODE_FTMEN_MASK;
343  FTM_DualChlConfig(instance, chl, kFTM_Combine, ENABLE);
344  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
345  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
346  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
347  FTM_DualChlConfig(instance, chl, kFTM_Sync, ENABLE);
348  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
349  FTM_InstanceTable[instance]->CONTROLS[chl].CnSC |= (FTM_CnSC_MSB_MASK|FTM_CnSC_MSA_MASK);
350  FTM_PWM_InvertPolarity(instance, chl, kFTM_PWM_HighTrue);
351  FTM_InstanceTable[instance]->SYNC = FTM_SYNC_CNTMIN_MASK|FTM_SYNC_CNTMAX_MASK;
352  FTM_InstanceTable[instance]->SYNC |= FTM_SYNC_SWSYNC_MASK;
353  break;
354  case kFTM_Mode_QuadratureDecoder:
355  FTM_InstanceTable[instance]->QDCTRL |= FTM_QDCTRL_QUADEN_MASK;
356  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;
357  FTM_InstanceTable[instance]->MODE |= FTM_MODE_WPDIS_MASK;
358  FTM_InstanceTable[instance]->MODE |= FTM_MODE_FTMEN_MASK;
359  FTM_DualChlConfig(instance, chl, kFTM_Combine, ENABLE);
360  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
361  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
362  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
363  FTM_DualChlConfig(instance, chl, kFTM_Sync, DISABLE);
364  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
365  break;
366  default:
367  break;
368  }

406 uint8_t FTM_PWM_QuickInit(uint32_t MAP, FTM_PWM_Mode_Type mode, uint32_t req)
407 {
408  uint8_t i;
409  uint32_t modulo;
410  uint32_t clock;
411  int32_t pres;
412  uint8_t ps = 0;
413  map_t * pq = (map_t*)&(MAP);
414  /* init pinmux */
415  for(i = 0; i < pq->pin_cnt; i++)
416  {
417  PORT_PinMuxConfig(pq->io, pq->pin_start + i, (PORT_PinMux_Type) pq->mux);
418  }
419  /* calc req and ps */
420  uint32_t min_val = 0xFFFF;
421  /* cal ps */
422  clock = GetClock(kBusClock);
423  pres = (clock/req)/FTM_MOD_MOD_MASK;
424  if((clock/req)/pres > FTM_MOD_MOD_MASK)
425  {
426  pres++;
427  }
428  for(i = 0; i < 7; i++)
429  {
430  if((ABS(pres - (1<<i))) < min_val)
431  {
432  ps = i;
433  min_val = ABS(pres - (1<<i));
434  }
435  }
436  if(pres > (1<<ps)) ps++;
437  if(ps > 7) ps = 7;
438  modulo = ((clock/(1<<ps))/req) - 1;
439  LIB_TRACE("FTM PWM input frequency:%dHz\r\n", req);
440  LIB_TRACE("FTM PWM input clk:%d\r\n", clock);
441  LIB_TRACE("FTM PWM ps:%d\r\n", pres);
442  LIB_TRACE("FTM PWM modulo:%d\r\n", modulo);
443  _FTM_InitBasic(pq->ip, modulo, (FTM_ClockDiv_Type)ps);
444  /* set FTM mode */
445  switch(mode)
446  {
447  case kPWM_EdgeAligned:
448  FTM_SetMode(pq->ip, pq->chl, kFTM_Mode_EdgeAligned);
449  break;
450  case kPWM_Combine:
451  FTM_SetMode(pq->ip, pq->chl, kFTM_Mode_Combine);
452  break;
453  case kPWM_Complementary:
454  FTM_SetMode(pq->ip, pq->chl, kFTM_Mode_Complementary);
455  break;
456  default:
457  LIB_TRACE("error in FTM_PWM_Init\r\n");
458  break;
459  }
460  /* set duty to 50% */
461  FTM_PWM_ChangeDuty(pq->ip, pq->chl, 5000);
462 
463  /* set FTM clock to system clock */
464  FTM_InstanceTable[pq->ip]->SC &= ~FTM_SC_CLKS_MASK;
465  FTM_InstanceTable[pq->ip]->SC |= FTM_SC_CLKS(1);
466 
467  /* set ps, this must be done after set modulo */
468  FTM_InstanceTable[pq->ip]->SC &= ~FTM_SC_PS_MASK;
469  FTM_InstanceTable[pq->ip]->SC |= FTM_SC_PS(ps);
470 
471  return pq->ip;
472 }

499 void FTM_PWM_ChangeDuty(uint32_t instance, uint8_t chl, uint32_t pwmDuty)
500 {
501  uint32_t cv = ((FTM_InstanceTable[instance]->MOD) * pwmDuty) / 10000;
502  /* combine mode */
503  if(FTM_InstanceTable[instance]->COMBINE & (FTM_COMBINE_COMBINE0_MASK|FTM_COMBINE_COMBINE1_MASK|FTM_COMBINE_COMBINE2_MASK|FTM_COMBINE_COMBINE3_MASK))
504  {
505  if(chl%2)
506  {
507  FTM_InstanceTable[instance]->CONTROLS[chl-1].CnV = 0;
508  FTM_InstanceTable[instance]->CONTROLS[chl].CnV = cv;
509  }
510  else
511  {
512  FTM_InstanceTable[instance]->CONTROLS[chl].CnV = 0;
513  FTM_InstanceTable[instance]->CONTROLS[chl+1].CnV = cv;
514  }
515  FTM_InstanceTable[instance]->PWMLOAD = 0xFFFFFFFF;
516  }
517  else
518  {
519  /* single chl */
520  FTM_InstanceTable[instance]->CONTROLS[chl].CnV = cv;
521  }
