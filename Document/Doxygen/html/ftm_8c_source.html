132  break;

222 static void FTM_DualChlConfig(uint32_t instance, uint8_t chl, FTM_DualChlConfig_Type mode, FunctionalState newState)
223 {
224  uint32_t mask = 0;
225  switch(mode)
226  {
227  case kFTM_Combine:
228  mask = FTM_COMBINE_COMBINE0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
229  break;
230  case kFTM_Complementary:
231  mask = FTM_COMBINE_COMP0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
232  break;
233  case kFTM_DualEdgeCapture:
234  mask = FTM_COMBINE_DECAPEN0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
235  break;
236  case kFTM_DeadTime:
237  mask = FTM_COMBINE_DTEN0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
238  break;
239  case kFTM_Sync:
240  mask = FTM_COMBINE_SYNCEN0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
241  break;
242  case kFTM_FaultControl:
243  mask = FTM_COMBINE_FAULTEN0_MASK << (get_chl_pair_index(chl) * FTM_COMBINE_CHAN_CTRL_WIDTH);
244  break;
245  default:
246  break;
247  }
248  (newState == ENABLE)?(FTM_InstanceTable[instance]->COMBINE |= mask):(FTM_InstanceTable[instance]->COMBINE &= ~mask);
249 }

254 static void FTM_SetMode(uint32_t instance, uint8_t chl, FTM_Mode_Type mode)
255 {

263  FTM_PWM_InvertPolarity(instance, chl, kFTM_PWM_HighTrue);
264  FTM_DualChlConfig(instance, chl, kFTM_Combine, DISABLE);
265  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
266  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
267  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
268  FTM_DualChlConfig(instance, chl, kFTM_Sync, DISABLE);
269  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
270  break;
271  case kFTM_Mode_CenterAligned:
272  FTM_InstanceTable[instance]->MODE &= ~FTM_MODE_FTMEN_MASK;
273  FTM_InstanceTable[instance]->QDCTRL &= ~FTM_QDCTRL_QUADEN_MASK;
274  FTM_InstanceTable[instance]->SC |= FTM_SC_CPWMS_MASK;
275  FTM_DualChlConfig(instance, chl, kFTM_Combine, DISABLE);
276  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
277  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
278  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);

280  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
281  break;
282  case kFTM_Mode_InputCapture:
283  /* all configuration on input capture */
284  FTM_InstanceTable[instance]->QDCTRL &= ~FTM_QDCTRL_QUADEN_MASK;
285  FTM_InstanceTable[instance]->MODE &= ~FTM_MODE_FTMEN_MASK;
286  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;

288  FTM_DualChlConfig(instance, chl, kFTM_Combine, DISABLE);
289  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
290  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
291  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
292  FTM_DualChlConfig(instance, chl, kFTM_Sync, DISABLE);
293  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
294  break;
295  case kFTM_Mode_Complementary:
296  FTM_InstanceTable[instance]->QDCTRL &= ~FTM_QDCTRL_QUADEN_MASK;
297  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;
298  FTM_InstanceTable[instance]->MODE |= FTM_MODE_WPDIS_MASK;
299  FTM_InstanceTable[instance]->MODE |= FTM_MODE_FTMEN_MASK;
300  FTM_DualChlConfig(instance, chl, kFTM_Combine, ENABLE);
301  FTM_DualChlConfig(instance, chl, kFTM_Complementary, ENABLE);
302  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
303  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
304  FTM_DualChlConfig(instance, chl, kFTM_Sync, ENABLE);
305  FTM_DualChlConfig(instance, chl, kFTM_FaultControl, DISABLE);
306  FTM_InstanceTable[instance]->CONTROLS[chl].CnSC |= (FTM_CnSC_MSB_MASK|FTM_CnSC_MSA_MASK);
307  FTM_PWM_InvertPolarity(instance, chl, kFTM_PWM_HighTrue);
308  FTM_InstanceTable[instance]->SYNC = FTM_SYNC_CNTMIN_MASK|FTM_SYNC_CNTMAX_MASK;
309  FTM_InstanceTable[instance]->SYNC |= FTM_SYNC_SWSYNC_MASK;
310  break;
311  case kFTM_Mode_Combine:
312  FTM_InstanceTable[instance]->QDCTRL &= ~FTM_QDCTRL_QUADEN_MASK;
313  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;
314  FTM_InstanceTable[instance]->MODE |= FTM_MODE_WPDIS_MASK;
315  FTM_InstanceTable[instance]->MODE |= FTM_MODE_FTMEN_MASK;
316  FTM_DualChlConfig(instance, chl, kFTM_Combine, ENABLE);
317  FTM_DualChlConfig(instance, chl, kFTM_Complementary, DISABLE);
318  FTM_DualChlConfig(instance, chl, kFTM_DualEdgeCapture, DISABLE);
319  FTM_DualChlConfig(instance, chl, kFTM_DeadTime, DISABLE);
320  FTM_DualChlConfig(instance, chl, kFTM_Sync, ENABLE);

323  FTM_PWM_InvertPolarity(instance, chl, kFTM_PWM_HighTrue);
324  FTM_InstanceTable[instance]->SYNC = FTM_SYNC_CNTMIN_MASK|FTM_SYNC_CNTMAX_MASK;
325  FTM_InstanceTable[instance]->SYNC |= FTM_SYNC_SWSYNC_MASK;

329  FTM_InstanceTable[instance]->SC &= ~FTM_SC_CPWMS_MASK;

338  break;

341  }

379 uint8_t FTM_PWM_QuickInit(uint32_t MAP, FTM_PWM_Mode_Type mode, uint32_t req)
380 {
381  uint8_t i;
382  uint32_t modulo;
383  uint32_t clock;
384  int32_t pres;
385  uint8_t ps = 0;
386  map_t * pq = (map_t*)&(MAP);
387  /* init pinmux */
388  for(i = 0; i < pq->pin_cnt; i++)
389  {
390  PORT_PinMuxConfig(pq->io, pq->pin_start + i, (PORT_PinMux_Type) pq->mux);
391  }
392  /* calc req and ps */
393  uint32_t min_val = 0xFFFF;
394  /* cal ps */
395  clock = GetClock(kBusClock);
396  pres = (clock/req)/FTM_MOD_MOD_MASK;
397  if((clock/req)/pres > FTM_MOD_MOD_MASK)
398  {
399  pres++;
400  }
401  for(i = 0; i < 7; i++)
402  {
403  if((ABS(pres - (1<<i))) < min_val)
404  {
405  ps = i;
406  min_val = ABS(pres - (1<<i));
407  }
408  }
409  if(pres > (1<<ps)) ps++;
410  if(ps > 7) ps = 7;
411  modulo = ((clock/(1<<ps))/req) - 1;
412  LIB_TRACE("FTM PWM input frequency:%dHz\r\n", req);
413  LIB_TRACE("FTM PWM input clk:%d\r\n", clock);
414  LIB_TRACE("FTM PWM ps:%d\r\n", pres);
415  LIB_TRACE("FTM PWM modulo:%d\r\n", modulo);
416  _FTM_InitBasic(pq->ip, modulo, (FTM_ClockDiv_Type)ps);
417  /* set FTM mode */
418  switch(mode)
419  {
420  case kPWM_EdgeAligned:
421  FTM_SetMode(pq->ip, pq->chl, kFTM_Mode_EdgeAligned);
422  break;
423  case kPWM_Combine:
424  FTM_SetMode(pq->ip, pq->chl, kFTM_Mode_Combine);
425  break;
426  case kPWM_Complementary:
427  FTM_SetMode(pq->ip, pq->chl, kFTM_Mode_Complementary);
428  break;
429  default:
430  LIB_TRACE("error in FTM_PWM_Init\r\n");
431  break;
432  }
433  /* set duty to 50% */
434  FTM_PWM_ChangeDuty(pq->ip, pq->chl, 5000);
435 
436  /* set FTM clock to system clock */
437  FTM_InstanceTable[pq->ip]->SC &= ~FTM_SC_CLKS_MASK;
438  FTM_InstanceTable[pq->ip]->SC |= FTM_SC_CLKS(1);
439 
440  /* set ps, this must be done after set modulo */
441  FTM_InstanceTable[pq->ip]->SC &= ~FTM_SC_PS_MASK;
442  FTM_InstanceTable[pq->ip]->SC |= FTM_SC_PS(ps);
443 
444  return pq->ip;
445 }

472 void FTM_PWM_ChangeDuty(uint32_t instance, uint8_t chl, uint32_t pwmDuty)
473 {
