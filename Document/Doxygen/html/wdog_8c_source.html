58 void WDOG_Init(WDOG_InitTypeDef* WDOG_InitStruct)
59 {
60  uint32_t clock;
61  uint32_t time_out;
62  clock = GetClock(kBusClock);
63  uint32_t wdag_value = 0x01D3u;
64  switch((uint32_t)WDOG_InitStruct->mode)
65  {
66  case kWDOG_Mode_Normal:
67  wdag_value &= ~WDOG_STCTRLH_WINEN_MASK;
68  break;
69  case kWDOG_Mode_Window:
70  wdag_value |= WDOG_STCTRLH_WINEN_MASK;
71  break;
72  default:
73  break;
74  }
75  WDOG_Unlock();
76  /* set timeout value */
77  time_out = ((clock/8)/1000)*(WDOG_InitStruct->timeOutInMs);
78  WDOG->TOVALH = (time_out & 0xFFFF0000)>>16;
79  WDOG->TOVALL = (time_out & 0x0000FFFF)>>0;
80  /* set window time value :timeout must greater then window time */
81  time_out = ((clock/8)/1000)*(WDOG_InitStruct->windowInMs);
82  WDOG->WINH = (time_out & 0xFFFF0000)>>16;
83  WDOG->WINL = (time_out & 0x0000FFFF)>>0;
84  WDOG->PRESC = WDOG_PRESC_PRESCVAL(7); // perscale = 8
85  /* enable wdog */
86  wdag_value |= WDOG_STCTRLH_WDOGEN_MASK;
87  WDOG->STCTRLH = wdag_value;
88 }
