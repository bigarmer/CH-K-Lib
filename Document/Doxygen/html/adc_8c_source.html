90 int32_t ADC_Calibration(uint32_t instance)
91 {
92  uint32_t i;
93  uint32_t PG, MG;
94 
95  // disable clock
96 
97  ADC_InstanceTable[instance]->SC3 |= ADC_SC3_CALF_MASK; /* Clear the calibration's flag */
98  ADC_InstanceTable[instance]->SC3 |= ADC_SC3_CAL_MASK; /* Enable the calibration */
99 
100  for(i=0;i<100000;i++);
101  //while(ADC_IsConversionCompleted(instance, 0)) {}; /* Wait conversion is competed */
102  if(ADC_InstanceTable[instance]->SC3 & ADC_SC3_CALF_MASK)
103  {
104  /* cal failed */
105  LIB_TRACE("ADC calibration failed\r\n");
106  return 1;
107  }
108  /* Calculate plus-side calibration */
109  PG = 0U;
110  PG += ADC_InstanceTable[instance]->CLP0;
111  PG += ADC_InstanceTable[instance]->CLP1;
112  PG += ADC_InstanceTable[instance]->CLP2;
113  PG += ADC_InstanceTable[instance]->CLP3;
114  PG += ADC_InstanceTable[instance]->CLP4;
115  PG += ADC_InstanceTable[instance]->CLPS;
116  PG = 0x8000U | (PG>>1U);
117 
118  /* Calculate minus-side calibration */
119  MG = 0U;
120  MG += ADC_InstanceTable[instance]->CLM0;
121  MG += ADC_InstanceTable[instance]->CLM1;
122  MG += ADC_InstanceTable[instance]->CLM2;
123  MG += ADC_InstanceTable[instance]->CLM3;
124  MG += ADC_InstanceTable[instance]->CLM4;
125  MG += ADC_InstanceTable[instance]->CLMS;
126  MG = 0x8000U | (MG>>1U);
127  ADC_InstanceTable[instance]->SC3 &= ~ADC_SC3_CAL_MASK; /* Disable the calibration */
128  ADC_InstanceTable[instance]->PG = PG;
129  ADC_InstanceTable[instance]->MG = MG;
130  return 0;
131 }

151 void ADC_Init(ADC_InitTypeDef* ADC_InitStruct)
152 {
153  /* enable clock gate */
154  *(uint32_t*)SIM_ADCClockGateTable[ADC_InitStruct->instance].addr |= SIM_ADCClockGateTable[ADC_InitStruct->instance].mask;
155 
156  /* do calibration */
157  //ADC_Calibration(ADC_InitStruct->instance);
158 
159  /* select clock */
160  //ADC_InstanceTable[ADC_InitStruct->instance]->CFG2 |= ADC_CFG2_ADACKEN_MASK;
161  //ADC_InstanceTable[ADC_InitStruct->instance]->CFG1 &= ~ADC_CFG1_ADICLK_MASK;
162  //ADC_InstanceTable[ADC_InitStruct->instance]->CFG1 |= ADC_CFG1_ADICLK(3);
163 
164  /* set clock configuration */
165  ADC_InstanceTable[ADC_InitStruct->instance]->CFG1 &= ~ADC_CFG1_ADIV_MASK;
166  ADC_InstanceTable[ADC_InitStruct->instance]->CFG1 |= ADC_CFG1_ADIV(ADC_InitStruct->clockDiv);
167 
168  /* long sample time */
169  ADC_InstanceTable[ADC_InitStruct->instance]->CFG1 |= ADC_CFG1_ADLPC_MASK;
170 
171  /* voltage reference */
172  ADC_InstanceTable[ADC_InitStruct->instance]->SC2 &= ~ADC_SC2_REFSEL_MASK;
173  ADC_InstanceTable[ADC_InitStruct->instance]->SC2 |= ADC_SC2_REFSEL(ADC_InitStruct->vref);
174 
175  /* resolutionMode */
176  ADC_InstanceTable[ADC_InitStruct->instance]->CFG1 &= ~(ADC_CFG1_MODE_MASK);
177  ADC_InstanceTable[ADC_InitStruct->instance]->CFG1 |= ADC_CFG1_MODE(ADC_InitStruct->resolutionMode);
178 
179  /* trigger mode */
180  (kADC_TriggerHardware == ADC_InitStruct->triggerMode)?(ADC_InstanceTable[ADC_InitStruct->instance]->SC2 |= ADC_SC2_ADTRG_MASK):(ADC_InstanceTable[ADC_InitStruct->instance]->SC2 &= ADC_SC2_ADTRG_MASK);
181 
182  /* if continues conversion */
183  (kADC_ContinueConversionEnable == ADC_InitStruct->continueMode)?(ADC_InstanceTable[ADC_InitStruct->instance]->SC3 |= ADC_SC3_ADCO_MASK):(ADC_InstanceTable[ADC_InitStruct->instance]->SC3 &= ~ADC_SC3_ADCO_MASK);
184 
185  /* if hardware average enabled */
186  switch(ADC_InitStruct->hardwareAveMode)
187  {
188  case kADC_HardwareAverageDisable:
189  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 &= ~ADC_SC3_AVGE_MASK;
190  break;
191  case kADC_HardwareAverage_4:
192  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 |= ADC_SC3_AVGE_MASK;
193  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 &= ~ADC_SC3_AVGS_MASK;
194  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 |= ADC_SC3_AVGS(0);
195  break;
196  case kADC_HardwareAverage_8:
197  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 |= ADC_SC3_AVGE_MASK;
198  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 &= ~ADC_SC3_AVGS_MASK;
199  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 |= ADC_SC3_AVGS(1);
200  break;
201  case kADC_HardwareAverage_16:
202  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 |= ADC_SC3_AVGE_MASK;
203  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 &= ~ADC_SC3_AVGS_MASK;
204  ADC_InstanceTable[ADC_InitStruct->instance]->SC3 |= ADC_SC3_AVGS(2);
