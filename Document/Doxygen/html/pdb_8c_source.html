32 static void _PDB_SetCounterPeriod(uint32_t srcClock, uint32_t timeInUs)
33 {
34  static const uint8_t MULT[4] = {1, 10, 20, 40};
35  uint32_t clkDiv,mult;
36  uint32_t i,j;
37  for(i=0;i<8;i++)
38  {
39  for(j=0;j<4;j++)
40  {
41  if((srcClock/1000000)*timeInUs/((1<<i)*MULT[j]) < 0xFFFF)
42  {
43  clkDiv = i;
44  mult = j;
45  break;
46  }
47  }
48  }
49 
50  LIB_TRACE("clkDiv:%d Mult:%d\r\n",clkDiv, mult);
51 
52  /* clk div */
53  PDB0->SC &= ~PDB_SC_PRESCALER_MASK;
54  PDB0->SC |= PDB_SC_PRESCALER(clkDiv);
55 
56  /* muti */
57  PDB0->SC &= ~PDB_SC_MULT_MASK;
58  PDB0->SC |= PDB_SC_MULT(mult);
59 
60  LIB_TRACE("MOD:%d\r\n", (srcClock/1000000)*timeInUs/((1<<clkDiv)*MULT[mult]));
61  PDB0->MOD = (srcClock/1000000)*timeInUs/((1<<clkDiv)*MULT[mult]);
62  PDB0->IDLY = (srcClock/1000000)*timeInUs/((1<<clkDiv)*MULT[mult]);
63 }
