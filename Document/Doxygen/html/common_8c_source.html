38 uint32_t GetClock(Clock_t clockName)
39 {
40  uint32_t clock = 0;
41  /* calualte MCGOutClock system_MKxxx.c must not modified */
42  SystemCoreClockUpdate();
43  clock = SystemCoreClock * MCGOUT_TO_CORE_DIVIDER;
44  switch (clockName)
45  {
46  case kCoreClock:
47  clock = clock / MCGOUT_TO_CORE_DIVIDER;
48  break;
49  case kBusClock:
50  clock = clock / MCGOUT_TO_BUS_DIVIDER;
51  break;
52  case kFlashClock:
53  clock = clock / MCGOUT_TO_FLASH_DIVIDER;
54  break;
55  case kMCGOutClock:
56  break;
57  default:
58  clock = 0;
59  }
60  return clock;
61 }

141 void DWT_DelayUs(uint32_t us)
142 {
143  uint32_t startts, endts, ts;
144  startts = DWT->CYCCNT;
145  ts = us * (SystemCoreClock /(1000*1000) );
146  endts = startts + ts;
147  if(endts > startts)
148  {
149  while(DWT->CYCCNT < endts);
150  }
151  else
152  {
153  while(DWT->CYCCNT > endts);
154  while(DWT->CYCCNT < endts);
155  }
